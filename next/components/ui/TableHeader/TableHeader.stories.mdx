# TableHeader

TableHeader

## Props

TableHeader accepts custom Props that are as follows:

- <b>className</b>: modifies or extends the styles for the component
- <b>header</b>: content of the component
- <b>field</b>: identificator of the column
- <b>sortable</b>: if true, column is sortable, otherwise not sortable
- <b>sortBy</b>: object, contains key and direction of column, by which is table sorted
- <b>onSort</b>: function, triggers sort function defined in table

export const Template = (args) => {
  const [sortBy, setSortBy] = React.useState({
    key: 'email',
    direction: 'desc',
  })
  return (
    <TableHeader
      header={args.header}
      field={args.field}
      sortable={args.sortable}
      sortBy={sortBy}
      onSort={() => {
        setSortBy({
          key: sortBy.key,
          direction: sortBy.direction === 'desc' ? 'asc' : 'desc',
        })
      }}
    />
  )
}
