# Accordion

Accordion

## Props

Accordion accepts custom Props that are as follows:

- <b>className</b>: modifies or extends the styles for the component
- <b>sections</b>: sections for the Accordion
- <b>activeSectiont</b>: key of current expanded section
- <b>onSelect</b>: function, accepts as parameter selected section as object
- <b>contents</b>: array of contents (components) of individual sections.

export const Template = (args) => {
  const [activeSection, setActiveSection] = React.useState('0')
  return (
    <Accordion
      className={args.className}
      tabs={args.tabs}
      tabVariant="justify-center"
      contents={args.contents}
      activeSection={activeSection}
      onSelect={(d) => setActiveSection(d)}
    />
  )
}
